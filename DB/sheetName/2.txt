  `login` varchar(25) DEFAULT NULL COMMENT '登录账号',
  `password` varchar(255) DEFAULT NULL COMMENT '用户密码的md5摘要',
  `login_salt` char(5) DEFAULT NULL COMMENT '10000 到 99999之间的随机数，加密密码时使用',
  `email` varchar(255) DEFAULT NULL COMMENT '用户email',
  `qq` varchar(15) DEFAULT NULL COMMENT 'qq',
  `phone` varchar(15) DEFAULT NULL COMMENT '手机号',
  `is_bindphone` tinyint(2) NOT NULL DEFAULT '0' COMMENT '是否绑定手机用户，0否，1：是',
  `role_id` int(2) DEFAULT NULL COMMENT '关联角色表',
  `parent_id` int(11) DEFAULT NULL COMMENT '上级ID',

  `ehr_parent_id` int(11) NOT NULL DEFAULT '0' COMMENT 'ehr的直线经理在业务平台的user_id',
  `ehr_parent_name` varchar(25) NOT NULL DEFAULT '' COMMENT '行政汇报上级姓名',
  `ehr_parent_ysb_no` char(9) NOT NULL DEFAULT '' COMMENT '行政汇报上级工号',
  `state` tinyint(1) DEFAULT NULL COMMENT '状态0待激活 1启用 2注销',
  `note` varchar(200) DEFAULT NULL,
  `bind_ysbuser_id` int(11) DEFAULT NULL COMMENT '绑定代表自己的药师帮店员版用户ID',
  `job_title` tinyint(1) NOT NULL DEFAULT '0' COMMENT '职称 0-未设置 1-试用期代表 2-正式代表',
  `ysb_no` char(9) NOT NULL DEFAULT '' COMMENT '工号',
  `ctime` int(11) NOT NULL DEFAULT '0' COMMENT '注册时间',
  `report_max` smallint(11) NOT NULL DEFAULT '120' COMMENT '业务平台报备药品数量上限',
  `renling_max` smallint(11) DEFAULT '150' COMMENT '认领药店上限，-1表示不限',
  `mtime` int(11) NOT NULL DEFAULT '0' COMMENT '修改时间',
  `last_login_time` datetime NOT NULL DEFAULT '1000-01-01 00:00:00' COMMENT '最后登录时间',
  `last_login_ip` varchar(50) NOT NULL DEFAULT '' COMMENT '最后登录ip',
  `commisson_type` int(11) NOT NULL DEFAULT '3' COMMENT '旗舰店提成类型,ts_salesmen_commission_type.id',
  `active_store_flag` tinyint(4) NOT NULL DEFAULT '0' COMMENT '0-未配置 1-活跃店数少 2-活跃店数多',
  `eb_role` tinyint(4) NOT NULL DEFAULT '0' COMMENT '电商运营组织角色，用于电商运营管理模块，参考字段E_BUSI_ORG_ROLE',
  `eb_note` varchar(100) NOT NULL DEFAULT '' COMMENT '组织角色备注',
  PRIMARY KEY (`id`),
  UNIQUE KEY `login_uk` (`login`) USING BTREE,
  KEY `parent_idx` (`parent_id`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=10544 DEFAULT CHARSET=utf8;
